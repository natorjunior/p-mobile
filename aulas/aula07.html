// components/TodoInput.js
import React, { useState, useContext } from 'react';
import { View, TextInput, Button, StyleSheet } from 'react-native';
import { TodoContext, ThemeContext } from '../contexts';

const TodoInput = () => {
  const [text, setText] = useState('');
  const { todos, setTodos } = useContext(TodoContext);
  const { theme } = useContext(ThemeContext);

  const addTodo = () => {
    if (text.trim()) {
      setTodos([...todos, { id: Date.now(), text, completed: false }]);
      setText('');
    }
  };

  return (
    <View style={styles.container}>
      <TextInput
        style={[
          styles.input,
          { backgroundColor: theme === 'light' ? '#fff' : '#333',
            color: theme === 'light' ? '#000' : '#fff' }
        ]}
        value={text}
        onChangeText={setText}
        placeholder="Nova tarefa..."
        placeholderTextColor={theme === 'light' ? '#666' : '#999'}
      />
      <Button title="Adicionar" onPress={addTodo} />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    padding: 20,
    flexDirection: 'row',
  },
  input: {
    flex: 1,
    marginRight: 10,
    padding: 10,
    borderWidth: 1,
    borderRadius: 5,
  },

Passo 4: Componente TodoList


// components/TodoList.js
import React, { useContext } from 'react';
import { View, Text, TouchableOpacity, StyleSheet } from 'react-native';
import { TodoContext, ThemeContext } from '../contexts';

const TodoList = () => {
  const { todos, setTodos } = useContext(TodoContext);
  const { theme } = useContext(ThemeContext);

  const toggleTodo = (id) => {
    setTodos(todos.map(todo =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    ));
  };

  const deleteTodo = (id) => {
    setTodos(todos.filter(todo => todo.id !== id));
  };

  return (
    <View style={styles.container}>
      {todos.map(todo => (
        <TouchableOpacity
          key={todo.id}
          style={[
            styles.todoItem,
            { backgroundColor: theme === 'light' ? '#f8f8f8' : '#333' }
          ]}
          onPress={() => toggleTodo(todo.id)}
          onLongPress={() => deleteTodo(todo.id)}
        >
          <Text style={[
            styles.todoText,
            { 
              color: theme === 'light' ? '#000' : '#fff',
              textDecorationLine: todo.completed ? 'line-through' : 'none'
            }
          ]}>
            {todo.text}
          </Text>
        </TouchableOpacity>
      ))}
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    padding: 20,
  },
  todoItem: {
    padding: 15,
    marginBottom: 10,
    borderRadius: 5,
    elevation: 2,
  },
  todoText: {
    fontSize: 16,
  },
});
