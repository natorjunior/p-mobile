import React, { useState, useEffect } from 'react';
import { View, Text, TextInput, Button } from 'react-native';

export default function TaskList() { 
  const [tasks, setTasks] = useState([]);
  const [input, setInput] = useState('');

  useEffect(() => { 
      console.log('Lista de tarefas atualizada:', tasks); 
  }, [tasks]);

  const addTask = () => { 
      if (input.trim()) { 
          setTasks([...tasks, input.trim()]); 
          setInput('');
      } 
  };

  return ( 
      <View style={{ padding: 20 }}> 
          <Text style={{ fontSize: 24, marginBottom: 10 }}>Minhas Tarefas</Text> 
          <TextInput 
              placeholder="Digite uma tarefa..." 
              value={input} 
              onChangeText={setInput}
              style={{ height: 40, borderColor: '#fff', borderWidth: 1, marginBottom: 10, color: 'white', paddingHorizontal: 10 }}
          /> 
          <Button 
              title="Adicionar" 
              onPress={addTask} 
          /> 
          {tasks.map((task, index) => ( 
              <Text key={index} style={{ marginTop: 10 }}>{task}</Text> 
          ))} 
      </View> 
  ); 
}
