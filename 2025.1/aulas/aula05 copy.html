Gerenciamento de Estado & useState

Utilize o hook useState para gerenciar o estado de forma dinâmica em seu componente. 

            
import React, { useState } from 'react';
import { View, Text, Button } from 'react-native';

export default function Counter() { 
    const [count, setCount] = useState(0);

    return ( 
        <View style={{ alignItems: 'center', padding: 20 }}> 
            <Text style={{ fontSize: 24, marginBottom: 10 }}>Contador: {count}</Text> 
            <Button 
                title="Incrementar" 
                onPress={() => setCount(count + 1)} 
            /> 
            <Button 
                title="Resetar" 
                onPress={() => setCount(0)} 
            /> 
        </View> 
    ); 
}


Introdução ao useEffect

O hook useEffect permite executar efeitos colaterais, como atualizar o console após mudanças de estado. 

            
import React, { useState, useEffect } from 'react';
import { View, Text, Button } from 'react-native';

export default function CounterWithEffect() { 
    const [count, setCount] = useState(0);

    useEffect(() => { 
        console.log('O contador foi atualizado:', count); 
    }, [count]);

    return ( 
        <View style={{ alignItems: 'center', padding: 20 }}> 
            <Text style={{ fontSize: 24, marginBottom: 10 }}>Contador: {count}</Text> 
            <Button 
                title="Incrementar" 
                onPress={() => setCount(count + 1)} 
            /> 
        </View> 
    ); 
}


Exercício Prático: Lista de Tarefas

Crie um componente que permita adicionar tarefas a uma lista utilizando useState e registre alterações com useEffect. 

            
import React, { useState, useEffect } from 'react';
import { View, Text, TextInput, Button } from 'react-native';

export default function TaskList() { 
    const [tasks, setTasks] = useState([]);
    const [input, setInput] = useState('');

    useEffect(() => { 
        console.log('Lista de tarefas atualizada:', tasks); 
    }, [tasks]);

    const addTask = () => { 
        if (input.trim()) { 
            setTasks([...tasks, input.trim()]); 
            setInput('');
        } 
    };

    return ( 
        <View style={{ padding: 20 }}> 
            <Text style={{ fontSize: 24, marginBottom: 10 }}>Minhas Tarefas</Text> 
            <TextInput 
                placeholder="Digite uma tarefa..." 
                value={input} 
                onChangeText={setInput}
                style={{ height: 40, borderColor: '#fff', borderWidth: 1, marginBottom: 10, color: 'white', paddingHorizontal: 10 }}
            /> 
            <Button 
                title="Adicionar" 
                onPress={addTask} 
            /> 
            {tasks.map((task, index) => ( 
                <Text key={index} style={{ marginTop: 10 }}>{task}</Text> 
            ))} 
        </View> 
    ); 
}
